\section{Tạo bảng và dữ liệu mẫu}
\subsection{Các câu lệnh hiện thực tất cả bảng dữ liệu }

\subsubsection{Tạo bảng KHOA}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `KHOA` (
	`Ma` INT PRIMARY KEY,
	`Ten` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci UNIQUE,
	`Email` VARCHAR(50) UNIQUE,
	`Nam thanh lap` DATE,
	`So dien thoai` VARCHAR(50)
	);	
\end{lstlisting}

\subsubsection{Tạo bảng TÁC GIẢ TÀI LIỆU}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `TAC GIA TAI LIEU` (
	`Ma tai lieu` INT,
	`Ten tac gia` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	PRIMARY KEY(`Ma tai lieu`,`Ten tac gia`)
	);	
\end{lstlisting}

\subsubsection{Tạo bảng GIẢNG VIÊN}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `GIANG VIEN` (
	`Ma EDUMEMBER` INT PRIMARY KEY,
	`Hoc vi` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	`Chuc danh` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	`Linh vuc` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	`So nam kinh nghiem` INT,
	`Ma khoa` INT NOT NULL
	);
\end{lstlisting}

\subsubsection{Tạo bảng MÔN HỌC}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `MON HOC` (
	`Ma` INT PRIMARY KEY,
	`Ten` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	`So tin chi` INT,
	`Mo ta` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci
	);
\end{lstlisting}

\subsubsection{Tạo bảng TL THUỘC VỀ}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `TL THUOC VE` (

	PRIMARY KEY (`Ma tai lieu`,`Ten loai tai lieu`)
	);

\subsubsection{Tạo bảng LOẠI TÀI LIỆU}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `LOAI TAI LIEU` (

	);
\end{lstlisting}

\subsubsection{Tạo bảng HỌC KÌ}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `HOC KI` (
	`Ten hoc ki` VARCHAR(50) PRIMARY KEY
	);
\end{lstlisting}

\subsubsection{Tạo bảng LỚP HỌC}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `LOP HOC` (
	`Ten hoc ki` VARCHAR(50),
	`Ma mon hoc` INT,
	`Ten lop` VARCHAR(50),
	`Ma giang vien` INT NOT NULL,
	`Thu` ENUM('2','3','4','5','6','7','CN'),
	`Gio hoc` TIME,
	`Dia diem` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	PRIMARY KEY (`Ten hoc ki`, `Ma mon hoc`, `Ten lop`)
	);
\end{lstlisting}

\subsubsection{Tạo bảng Tài liệu lớp học}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `Tai lieu lop hoc` (
	`Ten hoc ki` VARCHAR(50),
	`Ma mon hoc` INT,
	`Ten lop` VARCHAR(50),
	`Ma tai lieu` INT,
	`Ten` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	`Mo ta` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	PRIMARY KEY (`Ten hoc ki`, `Ma mon hoc`, `Ten lop`, `Ma tai lieu`)
	);
\end{lstlisting}

\subsubsection{Tạo bảng TLLH THUỘC VỀ}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `TLLH THUOC VE` (
	`Ten hoc ki` VARCHAR(50),
	`Ma mon hoc` INT,
	`Ten lop` VARCHAR(50),
	`Ma tai lieu lop hoc` INT,
	`Ten loai tai lieu` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	PRIMARY KEY (`Ten hoc ki`, `Ma mon hoc`, `Ten lop`, `Ma tai lieu lop hoc`, `Ten loai tai lieu`)
	);
\end{lstlisting}

\subsubsection{Tạo bảng HỌC}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `HOC` (
	`Ma sinh vien` INT,
	`Ten hoc ki` VARCHAR(50),
	`Ma mon hoc` INT,
	`Ten lop` VARCHAR(50),
	`Diem tong ket` INT,
	`Ngay dang ki` DATE,
	`Trang thai hoc` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	PRIMARY KEY (`Ma sinh vien`, `Ten hoc ki`, `Ma mon hoc`, `Ten lop`)
	);
\end{lstlisting}

\subsubsection{Tạo bảng BÀI TẬP}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `BAI TAP` (
	`Ten hoc ki` VARCHAR(50),
	`Ma mon hoc` INT,
	`Ten lop` VARCHAR(50),
	`Ma bai tap` INT,
	`Tuy chon` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	`Hinh thuc lam bai` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	`Ten bai` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	`Mo ta` VARCHAR(50) CHARACTER SET utf8mb4 collate utf8mb4_vietnamese_ci,
	`Thoi gian bat dau` DATE,
	`Thoi gian ket thuc` DATE,
	PRIMARY KEY (`Ten hoc ki`, `Ma mon hoc`, `Ten lop`, `Ma bai tap`)
	);
\end{lstlisting}

\subsubsection{Tạo bảng THAM GIA LÀM}
\begin{lstlisting}[language=SQL]
	CREATE TABLE IF NOT EXISTS `THAM GIA LAM` (
	`Ma sinh vien` INT,
	`Ten hoc ki` VARCHAR(50),
	`Ma mon hoc` INT,
	`Ten lop` VARCHAR(50),
	`Ma bai tap` INT,
	`Thoi gian nop bai` DATETIME,
	`Diem so` DECIMAL(10,2),
	PRIMARY KEY (`Ma sinh vien`, `Ten hoc ki`, `Ma mon hoc`, `Ten lop`, `Ma bai tap`)
	);
\end{lstlisting}
